%% SLIDE 2: MOTIVATION
\begin{frame}{Motivation}
\vspace{0.5cm}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
    \textbf{The Problem}
    \begin{itemize}
        \item Distributed caches: multi-tier systems
        \item Expensive to query each tier
        \item Need fast membership test
        \item Memory constraints
    \end{itemize}
    
    \vspace{0.5cm}
    
    \textbf{The Solution}
    \begin{itemize}
        \item Bloom filters (Bloom, 1970)
        \item Probabilistic data structure
        \item Space-efficient membership test
        \item Allows false positives, \textcolor{red}{never false negatives}
    \end{itemize}
\end{column}

\begin{column}{0.48\textwidth}
    \textbf{Real-World Usage}
    \begin{itemize}
        \item Google BigTable
        \item Amazon DynamoDB
        \item Apache Cassandra
        \item Akamai CDN
    \end{itemize}
    
    \vspace{0.5cm}
    
    \textbf{Research Question}
    \begin{center}
    \fcolorbox{darkblue}{lightgray}{%
    \parbox{0.9\textwidth}{
        \centering
        \textit{Do Bloom filters significantly reduce cache latency in distributed systems?}
    }}
    \end{center}
\end{column}
\end{columns}
\end{frame}

%% SLIDE 7: KEY RESULT 2 - THEORETICAL VALIDATION
\begin{frame}{Result 2: Theoretical Validation}
\vspace{0.5cm}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
    \textbf{Bloom's Formula (1970)}
    \begin{equation*}
    \text{FPR} = \left(1 - e^{-kn/m}\right)^k
    \end{equation*}
    
    where:
    \begin{itemize}
        \item $k$ = number of hash functions
        \item $n$ = elements inserted
        \item $m$ = bit array size
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{Optimal Parameters}
    \begin{align*}
    k^* &= \frac{m}{n} \ln(2) \\
    m^* &= -\frac{n \ln(p)}{(\ln 2)^2}
    \end{align*}
\end{column}

\begin{column}{0.48\textwidth}
    \textbf{Validation Results}
    \begin{itemize}
        \item Chi-square goodness-of-fit test
        \item \textit{p} $>$ 0.05 for all configurations
        \item \textbf{ACCEPT H$_0$}: Empirical = Theoretical
    \end{itemize}
    
    \vspace{0.3cm}
    
    \begin{block}{Conclusion}
    Implementation \textcolor{darkblue}{perfectly matches} 54-year-old theoretical predictions
    \end{block}
    
    \vspace{0.3cm}
    
    \textbf{Associations}
    \begin{itemize}
        \item Fill ratio $\leftrightarrow$ FPR: \textit{r} = 0.89 (\textit{p} = 0.041)
        \item m/n ratio $\leftrightarrow$ FPR: \textit{r} = 0.94 (\textit{p} = 0.019)
    \end{itemize}
\end{column}
\end{columns}
\end{frame}

%% SLIDE 9: REGRESSION MODEL
\begin{frame}{Result 3: Predictive Model}
\vspace{0.4cm}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
    \textbf{Linear Regression Model}
    \begin{equation*}
    \log(\text{FPR}) = \beta_0 + \beta_1 \left(\frac{m}{n}\right) + \varepsilon
    \end{equation*}
    
    \textbf{Estimation:} Ordinary Least Squares (OLS)
    
    \vspace{0.3cm}
    
    \textbf{Parameter Estimates}
    \begin{table}
    \centering
    \small
    \begin{tabular}{lcc}
    \toprule
    & \textbf{Estimate} & \textbf{p-value} \\
    \midrule
    $\beta_0$ & 6.99 & 0.100 \\
    $\beta_1$ & -1.40 & \textbf{0.002} \\
    \bottomrule
    \end{tabular}
    \end{table}
\end{column}

\begin{column}{0.48\textwidth}
    \textbf{Model Fit}
    \begin{itemize}
        \item R$^2$ = 0.41 (41\% variance)
        \item F-statistic = 13.12
        \item \textit{p} = 0.002 (significant)
    \end{itemize}
    
    \vspace{0.3cm}
    
    \begin{block}{Decision}
    \textbf{REJECT H$_0$} (\textit{p} $<$ 0.01) \\
    m/n ratio \textcolor{darkblue}{significantly predicts} FPR
    \end{block}
    
    \vspace{0.3cm}
    
    \textbf{ANOVA} (Workload Effects)
    \begin{itemize}
        \item $\eta^2$ = 0.46 (large effect)
        \item 46\% of latency variance explained
        \item Different workloads differ significantly
    \end{itemize}
\end{column}
\end{columns}
\end{frame}
